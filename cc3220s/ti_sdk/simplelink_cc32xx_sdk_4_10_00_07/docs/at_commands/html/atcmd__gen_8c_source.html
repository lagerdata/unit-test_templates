215 int32_t ATCmd_signalEvent(ATCmd_Event_t *pmsg)
216 {
217  struct mq_attr attr;
218  uint8_t doSend = 0;
219 
220  do
221  {
222  mq_getattr(ATCmd_rxQueue, &attr);
223  if (attr.mq_curmsgs < 10)
224  {
225  mq_send(ATCmd_rxQueue, (char *)pmsg, sizeof(ATCmd_Event_t), 0);
226  doSend = 1;
227  }
228  else
229  {
230  usleep(1000);
231  }
232  }while (!doSend);
233 
234  return 0;
235 }
